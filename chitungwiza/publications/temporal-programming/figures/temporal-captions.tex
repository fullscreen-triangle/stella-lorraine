\begin{figure*}[!htbp]

    \caption{\textbf{Composition Inflation Theorem:
    $T(n,d) = d\,(d+1)^{n-1}$ counts distinguishable
    labeled timing trajectories and grows exponentially in both
    the number of events $n$ and the dimension $d$.}
    (\textbf{A}) $T(n,d)$ versus $n$ on a logarithmic axis for
    $d = 1,\ldots,6$ (coloured lines).
    Each series is linear on the log scale, confirming the exponential
    growth law; the slope $(d+1)$ increases with $d$, spanning seven
    decades over $n = 1$ to $11$.
    (\textbf{B}) Three-dimensional surface of $\log_{10} T(n,d)$ over
    the grid $n \in [1,10]$, $d \in [1,8]$.
    The surface rises steeply in both dimensions; the curvature in
    $d$ reflects the $(d+1)^{n-1}$ factor compounding with $d$.
    (\textbf{C}) Growth ratio $T(n,d)/T(n{-}1,d)$ versus $n$ for
    $d = 2,\ldots,6$.
    Dashed horizontal lines at $d{+}1$ confirm that the ratio is
    constant and equal to $d{+}1$ for all $n \geq 2$, providing
    direct numerical verification of the recurrence relation
    $T(n,d) = (d{+}1)\,T(n{-}1,d)$.
    (\textbf{D}) Direct enumeration versus closed-form formula for
    $d = 3$, $n = 1,\ldots,7$ (blue and red bars, log scale).
    The enumeration sums $d^k$ over all integer compositions of $n$
    into $k$ parts; the formula $T(n,3) = 3 \cdot 4^{n-1}$ matches
    exactly, with bars overlapping at every value of $n$.}
    \label{fig:composition_inflation}
\end{figure*}

    \caption{\textbf{Monotone Time Coordinate and Structural Replay
    Resistance: $M(k)$ is non-decreasing by construction, and any
    replayed signal produces a $\Delta P$ distinguishable from the
    original.}
    (\textbf{A}) Monotone counter $M(k)$ versus signal index $k$
    (blue staircase with shaded area).
    Increments are drawn from $\{0,1,2\}$ at each step; the sequence
    is non-decreasing throughout, illustrating the constitutive
    asymmetry $t = M/f$ that forbids time reversal.
    (\textbf{B}) Timing deviation $\Delta P(k) = T_{\mathrm{ref}}(k)
    - t_{\mathrm{received}}(k)$ versus signal index.
    The signal fluctuates around zero with sub-cycle jitter; the
    shaded area tracks the sign of each deviation, providing the raw
    datum processed by timing cells.
    (\textbf{C}) Three-dimensional trajectory $(k,\,M_k,\,\Delta P_k)$
    in trajectory space.
    Red dots mark every tenth sample; the path is monotone in $M$
    regardless of $\Delta P$ sign, confirming that no trajectory can
    reverse its $M$-coordinate.
    (\textbf{D}) Scatter of replayed $\Delta P$ versus original $\Delta P$
    for 300 signal pairs.
    All replay values are shifted upward by a random delay
    $\delta \in [0.18,\,0.75]$~s, placing every point above the
    identity diagonal (dashed).
    A replayed signal is thus mapped to a different cell than the
    original, making the instruction structurally unexecutable.}
    \label{fig:monotone_replay}
\end{figure*}

    \caption{\textbf{COMPILE/EXECUTE Phase Separation Forced by the
    Receiver Uncertainty Principle
    $\sigma_K \cdot \sigma_Y \geq \hbar_R = \beta\tau$:
    trajectory accumulation and action firing are mutually exclusive
    phases.}
    (\textbf{A}) Antiphase oscillation of $\sigma_K$ (knowledge
    uncertainty, blue), $\sigma_Y$ (action uncertainty, red), and
    their product $\sigma_K \cdot \sigma_Y$ (green dotted) over four
    complete cycles.
    The product never drops below the floor $\hbar_R$ (grey dashed),
    and $\sigma_K$ is maximised precisely when $\sigma_Y$ is minimised,
    and vice versa.
    (\textbf{B}) Timeline of alternating COMPILE (blue) and EXECUTE
    (red) blocks over ten cycles.
    Block durations are drawn from realistic distributions; the two
    phases tile the time axis without overlap, illustrating the
    hard mutual exclusion enforced by the uncertainty principle.
    (\textbf{C}) Three-dimensional action-rate surface over
    $(\sigma_K,\,\sigma_Y) \in [0,1]^2$.
    The surface is nonzero only in the region $\sigma_K < 0.2$
    (deep EXECUTE phase) and grows with $\sigma_Y$, reflecting that
    actions fire at high confidence only after the trajectory has
    been fully resolved.
    (\textbf{D}) Action counts per block for 120 alternating blocks.
    COMPILE blocks (blue) fire zero actions by construction; EXECUTE
    blocks (red) fire a Poisson-distributed count $\sim\mathrm{Poi}(3.5)$,
    confirming that no action escapes the phase boundary.}
    \label{fig:phase_separation}
\end{figure*}
