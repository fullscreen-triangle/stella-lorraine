\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_01.png}
    \caption{\textbf{Kuramoto Order Parameter: numerical simulation
    confirms the mean-field prediction
    $R^* = \sqrt{1 - K_c/K}$ for $K > K_c$, with finite-size
    deviations scaling as $N^{-1/2}$.}
    (\textbf{A}) Ensemble order parameter $R_{\mathrm{ens}}$ versus
    coupling ratio $K/K_c$ for $N = 80$ Cauchy-distributed agents
    (red circles) against the mean-field curve
    $R^*(K) = \sqrt{1 - K_c/K}$ (blue line, valid for $K > K_c$).
    The vertical dashed line marks $K = K_c$; simulated values track
    the theoretical bifurcation, rising continuously from zero at
    the critical point.
    (\textbf{B}) Three-dimensional convergence of $R_{\mathrm{ens}}(t)$
    for four coupling ratios $K/K_c \in \{0.8, 1.2, 2.0, 3.5\}$
    (coloured trajectories).
    Sub-critical trajectories remain near zero; super-critical
    trajectories climb and plateau at a value that increases
    monotonically with $K/K_c$.
    (\textbf{C}) Phase distributions at the end of simulation for
    $K < K_c$ (blue bars) and $K > K_c$ (red bars) with $N = 150$.
    The sub-critical distribution is approximately uniform over
    $[0, 2\pi]$, while the super-critical distribution concentrates
    in a narrow peak, confirming the symmetry-breaking synchronisation
    transition.
    (\textbf{D}) Finite-size error $|R_{\mathrm{sim}} - R^*|$ versus
    $N$ on logarithmic axes at $K = 2.5\,K_c$ (blue circles) against
    the $1/\sqrt{N}$ reference (red dashed).
    Error decreases with $N$ at the expected rate, confirming that
    deviations from mean-field theory are pure finite-size fluctuations.}
    \label{fig:kuramoto_order}
\end{figure*}

\begin{figure*}[!htbp]

    \caption{\textbf{Critical Coupling Verification:
    $K_c = 2\sigma_\omega/\pi$ is confirmed numerically across a
    fivefold range of frequency spreads, and the ratio
    $K_c/\sigma_\omega$ is constant at $2/\pi$.}
    (\textbf{A}) $K_c$ versus $\sigma_\omega$ for
    $\sigma_\omega \in [0.2,\,3.0]$.
    Blue circles show the analytical prediction $2\sigma_\omega/\pi$;
    red triangles show the numerically detected transition point
    (first $K$ at which $R_{\mathrm{ens}}$ exceeds three times the
    finite-$N$ noise floor).
    Both series are collinear, confirming the linear
    $K_c$--$\sigma_\omega$ relationship.
    (\textbf{B}) Three-dimensional surface of $R_{\mathrm{ens}}(\sigma_\omega,\,K)$
    for $\sigma_\omega \in [0.3,\,2.5]$ and $K \in [0.1,\,5.0]$.
    The synchronisation manifold is the ridge where $R$ transitions
    from near-zero to positive; its projection onto the
    $(\sigma_\omega,\,K)$ plane follows the line $K = 2\sigma_\omega/\pi$.
    (\textbf{C}) Ratio $K_c/\sigma_\omega$ versus $\sigma_\omega$
    (blue circles) against the horizontal reference $2/\pi \approx 0.637$
    (red dashed).
    All points lie on the reference line, demonstrating that the
    critical coupling scales as a universal constant times
    $\sigma_\omega$.
    (\textbf{D}) $R_{\mathrm{ens}}$ versus $K/K_c$ for
    $N \in \{20, 40, 80, 150\}$ (coloured curves).
    The transition sharpens with increasing $N$; for large $N$ the
    curve approaches a step at $K/K_c = 1$, consistent with the
    discontinuous bifurcation predicted in the thermodynamic limit.}
    \label{fig:critical_coupling}
\end{figure*}

\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_04.png}
    \caption{\textbf{Zero-Overhead Phase Lock: coordination overhead
    vanishes discontinuously at $R_{\mathrm{ens}} = 0.95$, and the
    number of consensus messages avoided scales as $N(N-1)/2$.}
    (\textbf{A}) Coordination overhead versus $R_{\mathrm{ens}}$ (blue
    curve) with a vertical dashed line at $R = 0.95$ (red).
    Overhead decreases continuously below the threshold and drops
    to exactly zero above it; the green shading highlights the
    zero-overhead region $R \geq 0.95$.
    (\textbf{B}) Three-dimensional surface of per-agent overhead
    over $R_{\mathrm{ens}} \in [0,\,1]$ and $\log_{10} N \in [1.3,\,2.7]$.
    The surface descends to a flat zero plane for $R \geq 0.95$
    at all $N$, demonstrating that the overhead elimination is
    independent of federation size.
    (\textbf{C}) Empirical CDFs of time-to-phase-lock for
    $K/K_c \in \{1.5, 2.0, 3.0, 5.0\}$ (coloured curves) over
    40 independent realisations per condition.
    Higher coupling ratios produce faster lock: at $K = 5\,K_c$
    the CDF rises steeply near $t = 25$, while at $K = 1.5\,K_c$
    locking is slower, confirming that over-coupling accelerates
    convergence.
    (\textbf{D}) Log--log plot of consensus messages avoided versus
    $N$ (blue circles) against the $N^2/2$ reference (red dashed).
    The saving is exactly $N(N-1)/2$ per coordination event at
    $R \geq 0.95$, giving quadratic scaling with federation size.}
    \label{fig:zero_overhead}
\end{figure*}

\begin{figure*}[!htbp]

    \includegraphics[width=\textwidth]{figures/panel_05.png}
    \caption{\textbf{Composition Inflation and Federation Admission:
    $T(m,D) = D(D+1)^{m-1}$ counts distinguishable joint timing
    trajectories in a $D$-channel federation, and agents are admitted
    by marginal entropy gain $\Delta H_i > c_i$.}
    (\textbf{A}) $T(m,D)$ versus $m$ on a logarithmic axis for
    $D = 1,\ldots,6$ (coloured lines).
    Each series is linear on the log scale; the slope $(D+1)$
    increases with the number of sensor channels~$D$, quantifying
    the combinatorial richness of multi-channel swarm coordination.
    (\textbf{B}) Three-dimensional surface of $\log_{10} T(m,D)$
    over $m \in [1,9]$, $D \in [1,7]$.
    The surface rises steeply in both directions; equal-value
    contours are hyperbolas in the $(m,D)$ plane, reflecting the
    product structure $D \cdot (D+1)^{m-1}$.
    (\textbf{C}) Marginal entropy $\Delta H_i$ per agent (bar heights)
    against the per-agent cost threshold $c_i$ (black dashed curve)
    for 30 candidate agents.
    Blue bars (admitted) exceed their thresholds; red bars (rejected)
    do not.
    The admission rule $\Delta H_i > c_i$ selects agents that
    genuinely reduce federation uncertainty.
    (\textbf{D}) Federation entropy $H_{\mathrm{fed}}$ versus federation
    size $N$ (blue markers with shading).
    Growth is sub-linear ($H \propto \log N$), reflecting diminishing
    returns from additional agents: each new member adds less marginal
    information than the last.}
    \label{fig:composition_federation}
\end{figure*}

    \caption{\textbf{Multi-Sensor Federation and Graceful Decoherence:
    the federation sustains phase lock under a growing noise perturbation,
    transitions through successive regimes as $R_{\mathrm{ens}}$ decays,
    and decoherence is detected at a configurable threshold.}
    (\textbf{A}) $R_{\mathrm{ens}}$ versus time under progressive
    decoherence (blue curve).
    The system starts in the coherent regime ($R \approx 0.8$) after
    warm-up; a slowly increasing noise injection drives $R$ downward,
    passing through the regime boundaries at $0.80$, $0.50$, and $0.30$

    rather than abrupt failure.
    (\textbf{B}) Three-dimensional phase-difference matrix
    $|\phi_i - \phi_j| \bmod 2\pi$ at the end of a super-critical
    simulation ($K = 2.5\,K_c$, $N = 20$).
    The surface is nearly flat with small corrugations, confirming
    that all agent pairs are tightly phase-locked with
    $|\phi_i - \phi_j| \ll 1$ rad.
    (\textbf{C}) Density-normalised $\Delta P$ histograms from four
    heterogeneous sensor types (temperature, pressure, light,
    vibration) plotted on a common axis.
    Each domain contributes a distinct but overlapping distribution,
    demonstrating that diverse physical modalities are composable in
    a unified $\Delta P$ space via the multi-domain transduction
    framework.
    (\textbf{D}) Running $R_{\mathrm{ens}}(t)$ during a decoherence
    event (blue curve) with the detection threshold at $R = 0.50$
    (red dashed line).
    The trajectory drops below threshold at a well-defined time
    (red dot), at which point the federation scheduler can trigger
    a re-synchronisation protocol; the detection latency is bounded
    by $m^* = \lceil 1/\varepsilon \rceil$ timing events.}
    \label{fig:decoherence_detection}
\end{figure*}
